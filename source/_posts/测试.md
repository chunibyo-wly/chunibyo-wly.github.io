## 二级标题
```cpp
class Solution {
public:
    vector<int> twoSum(vector<int>& nums, int target) {
        map<int, int> m;
        vector<int> v;
        for(int i = 0; i < nums.size(); ++i) {
            m[nums[i]] = i;
        }
        for(int i = 0; i < nums.size(); ++i) {
            cout << target - nums[i] << ' ' << m[nums[i]] <<endl;
            if(m.find(target - nums[i]) != m.end() && m[target - nums[i]] != i) {
                v.push_back(i);
                v.push_back(m[target - nums[i]]);
                break;
            }
        }
        return v;
    }
};
```
